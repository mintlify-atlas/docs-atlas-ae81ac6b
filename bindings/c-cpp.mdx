ZBar provides both a C API and C++ wrapper classes for barcode scanning. The C API is the foundational interface, while the C++ wrappers provide an object-oriented interface with automatic memory management.

## Installation

### Using pkg-config

The recommended way to compile applications with ZBar is using `pkg-config`:

```bash
gcc -o myapp myapp.c $(pkg-config --cflags --libs zbar)
```

For C++ applications:

```bash
g++ -o myapp myapp.cpp $(pkg-config --cflags --libs zbar)
```

### Manual Compilation

If pkg-config is not available:

```bash
gcc -o myapp myapp.c -lzbar
g++ -o myapp myapp.cpp -lzbar
```

## C API

The C API is defined in `zbar.h`. Include it in your application:

```c
#include <zbar.h>
```

### Core Components

#### Image Scanner

The image scanner processes 2-D images to detect barcodes.

```c
#include <zbar.h>
#include <stdio.h>

// Create scanner
zbar_image_scanner_t *scanner = zbar_image_scanner_create();

// Configure (optional)
zbar_image_scanner_set_config(scanner, ZBAR_QRCODE, ZBAR_CFG_ENABLE, 1);

// Create image
zbar_image_t *image = zbar_image_create();
zbar_image_set_format(image, zbar_fourcc('Y','8','0','0'));
zbar_image_set_size(image, width, height);
zbar_image_set_data(image, raw_data, width * height, NULL);

// Scan image
int n = zbar_scan_image(scanner, image);

// Extract results
const zbar_symbol_t *symbol = zbar_image_first_symbol(image);
for(; symbol; symbol = zbar_symbol_next(symbol)) {
    zbar_symbol_type_t type = zbar_symbol_get_type(symbol);
    const char *data = zbar_symbol_get_data(symbol);
    printf("decoded %s symbol: %s\n",
           zbar_get_symbol_name(type), data);
}

// Cleanup
zbar_image_destroy(image);
zbar_image_scanner_destroy(scanner);
```

#### Processor

High-level interface for video processing:

```c
zbar_processor_t *proc = zbar_processor_create(1); // 1 = threaded

// Initialize with video device
zbar_processor_init(proc, "/dev/video0", 1); // 1 = enable display

// Configure
zbar_processor_set_config(proc, ZBAR_NONE, ZBAR_CFG_ENABLE, 1);

// Set callback
void my_handler(zbar_image_t *image, const void *userdata) {
    // Process results
}
zbar_processor_set_data_handler(proc, my_handler, NULL);

// Start scanning
zbar_processor_set_active(proc, 1);
zbar_processor_user_wait(proc, -1);

// Cleanup
zbar_processor_destroy(proc);
```

### Symbol Data Extraction

```c
const zbar_symbol_t *symbol = zbar_image_first_symbol(image);

for(; symbol; symbol = zbar_symbol_next(symbol)) {
    // Type and data
    zbar_symbol_type_t type = zbar_symbol_get_type(symbol);
    const char *data = zbar_symbol_get_data(symbol);
    unsigned int len = zbar_symbol_get_data_length(symbol);
    
    // Quality and confidence
    int quality = zbar_symbol_get_quality(symbol);
    int count = zbar_symbol_get_count(symbol);
    
    // Location polygon
    unsigned int n_points = zbar_symbol_get_loc_size(symbol);
    for(unsigned int i = 0; i < n_points; i++) {
        int x = zbar_symbol_get_loc_x(symbol, i);
        int y = zbar_symbol_get_loc_y(symbol, i);
    }
    
    // Orientation
    zbar_orientation_t orient = zbar_symbol_get_orientation(symbol);
}
```

## C++ API

The C++ API provides object-oriented wrappers with automatic resource management.

```cpp
#include <zbar.h>
using namespace zbar;
```

### Image Scanner

```cpp
#include <zbar.h>
#include <iostream>

using namespace zbar;

int main() {
    // Create scanner
    ImageScanner scanner;
    
    // Configure
    scanner.set_config(ZBAR_QRCODE, ZBAR_CFG_ENABLE, 1);
    
    // Create and configure image
    Image image(width, height, "Y800", raw_data, width * height);
    
    // Scan
    int n = scanner.scan(image);
    
    // Extract results
    for(Image::SymbolIterator symbol = image.symbol_begin();
        symbol != image.symbol_end();
        ++symbol) {
        std::cout << "decoded " << symbol->get_type_name()
                  << " symbol: " << symbol->get_data() << std::endl;
    }
    
    return 0;
}
```

### Processor with Callback

```cpp
class MyHandler : public Image::Handler {
public:
    void image_callback(Image &image) {
        for(Image::SymbolIterator symbol = image.symbol_begin();
            symbol != image.symbol_end();
            ++symbol) {
            std::cout << symbol->get_type_name() << ": "
                      << symbol->get_data() << std::endl;
        }
    }
};

int main() {
    Processor proc(true); // threaded
    proc.init("/dev/video0", true); // enable display
    proc.set_config(ZBAR_NONE, ZBAR_CFG_ENABLE, 1);
    
    MyHandler handler;
    proc.set_handler(handler);
    
    proc.set_active(true);
    proc.user_wait();
    
    return 0;
}
```

## Image Formats

ZBar requires grayscale image data. Common formats:

- **Y800** or **GRAY**: 8-bit grayscale (recommended)
- **JPEG**: Automatically converted to grayscale
- **RGB**: Can be converted using `zbar_image_convert()`

### Format Conversion

<CodeGroup>

```c C API
zbar_image_t *gray = zbar_image_convert(rgb_image, 
                                        zbar_fourcc('Y','8','0','0'));
```

```cpp C++ API
Image gray = rgb_image.convert("Y800");
```

</CodeGroup>

## Video Capture

### C API

```c
zbar_video_t *video = zbar_video_create();
zbar_video_open(video, "/dev/video0");
zbar_video_request_size(video, 640, 480);
zbar_video_init(video, zbar_fourcc('Y','8','0','0'));
zbar_video_enable(video, 1);

while(1) {
    zbar_image_t *image = zbar_video_next_image(video);
    zbar_scan_image(scanner, image);
    // Process results
    zbar_image_destroy(image);
}

zbar_video_destroy(video);
```

### C++ API

```cpp
Video video;
video.open("/dev/video0");
video.request_size(640, 480);
video.init("Y800");
video.enable(true);

while(true) {
    Image image = video.next_image();
    scanner.scan(image);
    // Process results
}
```

## Threading

Both APIs support threaded operation:

```c
// C: Create threaded processor
zbar_processor_t *proc = zbar_processor_create(1);
```

```cpp
// C++: Create threaded processor
Processor proc(true);
```

Threaded processors handle video capture and display in background threads, improving responsiveness.

## Memory Management

<Warning>
In C, you must manually destroy all created objects. In C++, destructors handle cleanup automatically.
</Warning>

### C API

```c
// Always destroy created objects
zbar_image_destroy(image);
zbar_image_scanner_destroy(scanner);
zbar_processor_destroy(proc);
zbar_video_destroy(video);
```

### C++ API

```cpp
// Automatic cleanup via destructors
{
    ImageScanner scanner;
    Image image;
    // Automatically destroyed when scope exits
}
```

## Reference Counting

Images and symbols use reference counting:

```c
// Increment reference count
zbar_image_ref(image, 1);
zbar_symbol_ref(symbol, 1);

// Decrement reference count
zbar_image_ref(image, -1);
zbar_symbol_ref(symbol, -1);
```

## Complete Example

<CodeGroup>

```c scan.c
#include <zbar.h>
#include <stdio.h>
#include <stdlib.h>

int main(int argc, char **argv) {
    // Create scanner
    zbar_image_scanner_t *scanner = zbar_image_scanner_create();
    zbar_image_scanner_set_config(scanner, 0, ZBAR_CFG_ENABLE, 1);
    
    // Process image file (simplified)
    zbar_image_t *image = zbar_image_create();
    zbar_image_set_format(image, zbar_fourcc('Y','8','0','0'));
    zbar_image_set_size(image, width, height);
    zbar_image_set_data(image, raw_data, width * height, NULL);
    
    // Scan
    int n = zbar_scan_image(scanner, image);
    printf("Found %d symbols\n", n);
    
    // Print results
    const zbar_symbol_t *symbol = zbar_image_first_symbol(image);
    for(; symbol; symbol = zbar_symbol_next(symbol)) {
        printf("%s: %s\n",
               zbar_get_symbol_name(zbar_symbol_get_type(symbol)),
               zbar_symbol_get_data(symbol));
    }
    
    // Cleanup
    zbar_image_destroy(image);
    zbar_image_scanner_destroy(scanner);
    
    return 0;
}
```
